template <class T>
    T expon(T n, T e) {
    T c = 1;
    for(int k = 0; k < abs(e); k++)
        c *= n;
    c = (e < 0) ? 1/c : c;
    return c;
}

template <class T>
    int get_bit(T x) {
    int i = 0, b = 0, c = 1;
    bool a;
    //for (i=0; i < 8; i++) {
    while (i < sizeof(x) * 8) {
        a = ((1<<i) & x)/(1<<i);
        for (int k=0; k<i; k++)
            c *= 10;
        b += (c * a);
        c = 1;
        i++;
    }
    return b;
}
